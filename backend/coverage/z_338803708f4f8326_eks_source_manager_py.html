148 def get_connector_processor(self, eks_connector, **kwargs): 

149 generated_credentials = generate_credentials_dict(eks_connector.type, eks_connector.keys) 

150 if 'region' in kwargs: 

151 generated_credentials['region'] = kwargs.get('region') 

152 

153 api_processor = EKSApiProcessor(**generated_credentials) 

154 

155 if 'cluster_name' in kwargs and kwargs['cluster_name']: 

156 client = kwargs.get('client', 'api') 

157 if 'native_connection' not in kwargs or not kwargs['native_connection']: 

158 return api_processor.eks_get_api_instance(cluster_name=kwargs['cluster_name'], client=client) 

159 else: 

160 instance = api_processor.eks_get_api_instance(cluster_name=kwargs['cluster_name'], client=client) 

161 eks_host = instance.api_client.configuration.host 

162 token = instance.api_client.configuration.api_key.get('authorization') 

163 ssl_ca_cert_path = instance.api_client.configuration.ssl_ca_cert 

164 return KubectlApiProcessor(api_server=eks_host, token=token, ssl_ca_cert_path=ssl_ca_cert_path) 

165 else: 

166 return api_processor

394 def execute_kubectl_command(self, time_range: TimeRange, eks_task: Eks, 

395 eks_connector: ConnectorProto) -> PlaybookTaskResult: 

396 try: 

397 if not eks_connector: 

398 raise Exception("Task execution Failed:: No EKS source found") 

399 

400 eks_command = eks_task.kubectl_command 

401 

402 region = eks_command.region.value 

403 cluster_name = eks_command.cluster.value 

404 

405 command_str = eks_command.command.value 

406 commands = command_str.split('\n') 

407 try: 

408 outputs = {} 

409 kubectl_client = self.get_connector_processor(eks_connector, region=region, cluster_name=cluster_name, 

410 client='api', native_connection=True) 

411 for command in commands: 

412 command_to_execute = command 

413 output = kubectl_client.execute_command(command_to_execute) 

414 outputs[command] = output 

415 

416 command_output_protos = [] 

417 for command, output in outputs.items(): 

418 bash_command_result = BashCommandOutputResult.CommandOutput( 

419 command=StringValue(value=command), 

420 output=StringValue(value=output) 

421 ) 

422 command_output_protos.append(bash_command_result) 

423 

424 return PlaybookTaskResult( 

425 source=self.source, 

426 type=PlaybookTaskResultType.BASH_COMMAND_OUTPUT, 

427 bash_command_output=BashCommandOutputResult( 

428 command_outputs=command_output_protos 

429 ) 

430 ) 

431 except Exception as e: 

432 raise Exception(f"Error while executing EKS kubectl task: {e}") 

433 except Exception as e: 

434 raise Exception(f"Error while executing EKS kubectl task: {e}")
