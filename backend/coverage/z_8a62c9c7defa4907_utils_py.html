24def timeseries_result_to_df(result: TimeseriesResult): 

25 data = [] 

26 for timeseries in result.labeled_metric_timeseries: 

27 legend_label = '' 

28 for label in timeseries.metric_label_values: 

29 legend_label += f'{label.name.value}={label.value.value}__' 

30 if legend_label and legend_label[-2:] == '__': 

31 legend_label = legend_label[:-2] 

32 for datapoint in timeseries.datapoints: 

33 data.append({ 

34 'Timestamp': pd.Timestamp(datapoint.timestamp, unit='ms').strftime('%H:%M'), 

35 'Value': datapoint.value.value, 

36 'Label': legend_label, 

37 'Unit': timeseries.unit.value 

38 }) 

39 

40 return pd.DataFrame(data)
