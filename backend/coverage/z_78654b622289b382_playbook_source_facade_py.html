41class PlaybookSourceFacade: 

42 

43 def __init__(self): 

44 self._map = {} 

45 

46 def register(self, source: Source, manager: PlaybookSourceManager): 

47 self._map[source] = manager 

48 

49 def get_source_manager(self, source: Source): 

50 if source not in self._map: 

51 raise ValueError(f'No executor found for source: {source}') 

52 return self._map.get(source) 

53 

54 def get_source_options(self, account_id) -> [PlaybookSourceOptions]: 

55 source_options: [PlaybookSourceOptions] = [] 

56 for source, manager in self._map.items(): 

57 connector_options: [PlaybookSourceOptions.ConnectorOption] = [] 

58 try: 

59 connector_protos: [ConnectorProto] = manager.get_active_connectors(account_id) 

60 for cp in connector_protos: 

61 connector_options.append(PlaybookSourceOptions.ConnectorOption(connector_id=cp.id, 

62 display_name=cp.display_name)) 

63 except Exception as e: 

64 logger.error(f'Error while getting connector options for source: {str(e)}') 

65 st_map = manager.get_task_type_callable_map() 

66 all_task_types = [] 

67 task_proto = manager.task_proto 

68 for task_type, task_info in st_map.items(): 

69 display_name = task_info['display_name'] 

70 task_type_name = task_proto.TaskType.Name(task_type) 

71 task_type_category = task_info.get('category', 'Others') 

72 model_types = task_info['model_types'] 

73 model_options: [PlaybookSourceOptions.TaskTypeOption.SourceModelTypeMap] = [] 

74 task_result_type = task_info.get('result_type', PlaybookTaskResultType.UNKNOWN) 

75 form_fields = task_info.get('form_fields', []) 

76 for m in model_types: 

77 model_type_display_name = model_type_display_name_maps.get(m, SourceModelType.Name(m)) 

78 model_options.append( 

79 PlaybookSourceOptions.TaskTypeOption.SourceModelTypeMap(model_type=m, display_name=StringValue( 

80 value=model_type_display_name))) 

81 all_task_types.append( 

82 PlaybookSourceOptions.TaskTypeOption(display_name=StringValue(value=display_name), 

83 task_type=StringValue(value=task_type_name), 

84 category=StringValue(value=task_type_category), 

85 supported_model_types=model_options, 

86 result_type=task_result_type, 

87 form_fields=form_fields)) 

88 display_name = integrations_connector_type_display_name_map.get(source, Source.Name(source)) 

89 source_options.append(PlaybookSourceOptions(source=source, 

90 display_name=StringValue(value=display_name), 

91 supported_task_type_options=all_task_types, 

92 connector_options=connector_options)) 

93 return source_options 

94 

95 def execute_task(self, account_id, time_range, global_variable_set: Struct, 

96 task: PlaybookTask) -> PlaybookTaskResult: 

97 source = task.source 

98 if source not in self._map: 

99 raise ValueError(f'No executor found for source: {source}') 

100 manager = self._map[source] 

101 try: 

102 return manager.execute_task(account_id, time_range, global_variable_set, task) 

103 except Exception as e: 

104 logger.error(f'Error while executing task: {str(e)}') 

105 return PlaybookTaskResult(error=StringValue(value=str(e))) 

106 

107 def test_source_connection(self, source_connection: ConnectorProto): 

108 source = source_connection.type 

109 if source not in self._map: 

110 return False, f'No executor found for source: {source}' 

111 manager = self._map[source] 

112 try: 

113 return manager.test_connector_processor(source_connection), None 

114 except Exception as e: 

115 logger.error(f'Error while testing source connection: {str(e)}') 

116 return False, str(e)
